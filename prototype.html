claim_service.pyPython
from decimal import Decimal, ROUND_HALF_UP
from dataclasses import dataclass
from enum import Enum

class ClaimStatus(Enum):
    APPROVED = "A"
    DENIED   = "D"
    PENDING  = "P"

@dataclass
class ClaimRecord:
    claim_id: str
    policy_number: str
    claim_amount: Decimal
    deductible: Decimal
    coverage_limit: Decimal
    status: ClaimStatus = ClaimStatus.PENDING
    payout: Decimal = Decimal("0")

class ClaimProcessor:
    def process(self, c):
        err = self._validate(c)
        if err:
            c.status = ClaimStatus.DENIED
            return c
        # BR-003
        net = (c.claim_amount - c.deductible
              ).quantize(Decimal("0.01"),
                         rounding=ROUND_HALF_UP)
        # BR-004
        c.payout = min(net, c.coverage_limit)
        # BR-005
        c.status = ClaimStatus.APPROVED
        return c

    def _validate(self, c):
        if c.claim_amount > c.coverage_limit:
            return 1001  # BR-001
        if not c.policy_number.strip():
            return 1002  # BR-002
        return None
